360  /**
361  * Set period of timer
362  *
363  * @param[in] period Period in seconds as a float
364  *
365  * @note Will enable and adjust the pre-scaler to appropriate value.\n
366  * The clock source should be selected by setClock() before using this function.
367  *
368  * @return true => success, false => failed to find suitable values for PBYP & PRESCALE
369  */
370  static ErrorCode setPeriod(float period) {
371  float inputClock = Info::getInputClockFrequency();
372  int prescaleFactor=1;
373  uint32_t prescalerValue=0;
374  while (prescalerValue<=16) {
375  float clock = inputClock/prescaleFactor;
376  uint32_t mod = round(period*clock);
377  if (mod < MINIMUM_RESOLUTION) {
378  // Too short a period for 1% resolution
379  return setErrorCode(E_TOO_SMALL);
380  }
381  if (mod <= 65535) {
382  // Disable LPTMR before prescale change
383  uint32_t csr = lptmr->CSR;
384  lptmr->CSR = 0;
385  __DSB();
386  lptmr->CMR = mod;
387  lptmr->PSR = (lptmr->PSR & ~(LPTMR_PSR_PRESCALE_MASK|LPTMR_PSR_PBYP_MASK))|LPTMR_PSR_PRESCALE(prescalerValue-1)|LPTMR_PSR_PBYP(prescalerValue==0);
388  lptmr->CSR = csr;
389  return E_NO_ERROR;
390  }
391  prescalerValue++;
392  prescaleFactor <<= 1;
393  }
394  // Too long a period
395  return setErrorCode(E_TOO_LARGE);
396  }
